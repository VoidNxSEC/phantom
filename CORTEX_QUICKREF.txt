🐍 ANÁLISE EM PYTHON
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

import json
from collections import Counter

# Carregar insights
insights = []
with open('insights.jsonl') as f:
    for line in f:
        insights.append(json.loads(line))

# Top keywords
keywords = []
for doc in insights:
    for theme in doc['themes']:
        keywords.extend(theme['keywords'])

print("Top 10 keywords:")
for kw, count in Counter(keywords).most_common(10):
    print(f"  {kw}: {count}")

# High confidence extractions
high_conf = [d for d in insights if d['extraction_confidence'] == 'high']
print(f"\\nHigh confidence: {len(high_conf)}/{len(insights)}")

# Categories de learnings
categories = Counter()
for doc in insights:
    for learning in doc['learnings']:
        categories[learning['category']] += 1

print("\\nLearning categories:", dict(categories))
